bool hanoi_swap(int start, int middle, int end, int rings) {
    print(rings);
    if ((rings == 0)) return true;
    auto pass1 = hanoi_swap(start, end, middle, rings - 1);
    print(rings);
    auto pass2 = hanoi_swap(middle, start, end, rings - 1);
    return pass1 && pass2;
}

bool hanoi(int number) {
