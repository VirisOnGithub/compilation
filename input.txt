bool hanoi_swap(int start, int middle, int end, int rings) {
    int line = 11111;
    if ((rings == 0)) return true;
    auto pass1 = hanoi_swap(start, end, middle, rings - 1);
    print(line);
    print(start);
    print(end);
    auto pass2 = hanoi_swap(middle, start, end, rings - 1);
    return pass1 && pass2;
}

bool hanoi(int number) {
    auto pass = hanoi_swap(0, 1, 2, number);
    return pass;
}

bool recursive(int number) {
    if ((number == 0)) return true;
    int line = 11111;
    int var = 5 - number;
    print(line);
    print(var);
    auto pass = recursive(number - 1);
    print(line);

    print(number);
    print(var);
    return pass;
}
